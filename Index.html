function update() { // Horizontal movement if(keys['a'] || keys['ArrowLeft']) player.x -= player.speed; // optional: keep A/left arrow if(keys['d'] || keys['ArrowRight']) player.x += player.speed; // Vertical movement if(keys['s'] || keys['ArrowDown']) { // Optional: crouch or drop through platforms // player.y += player.speed; // Only if you want S to move down } // Jump with W if(keys['w'] && player.grounded) { player.velocityY = -player.jumpPower; player.grounded = false; } // Apply gravity player.velocityY += gravity; player.y += player.velocityY; // Collision detection player.grounded = false; platforms.forEach(platform => { if(player.x < platform.x + platform.width && player.x + player.width > platform.x && player.y < platform.y + platform.height && player.y + player.height > platform.y) { if(player.velocityY > 0) { player.y = platform.y - player.height; player.velocityY = 0; player.grounded = true; } } }); }
